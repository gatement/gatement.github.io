setup kubernetes

  1. install CentOS 7
  2. config OS 
        systemctl disable firewalld.service
    systemctl stop firewalld.service

    # vi /etc/selinux/config
    SELINUX=disabled
  3. install softwares 
        yum install kubernetes etcd flannel docker
  4. config flannel 
        etcdctl set /atomic.io/network/config '{ "Network": "192.168.0.0/16" }'
