iptables

examples 

iptables -P INPUT ACCEPT
iptables -F
iptables -A INPUT -s 127.0.0.1 -p tcp --dport 9200 -j ACCEPT # localhost
iptables -A INPUT -p tcp --dport 9200 -j DROP

list rules 

iptables -L

connection count limit 

iptables -I INPUT -p tcp --dport 10000 -m connlimit --connlimit-above 2 -j DROP
