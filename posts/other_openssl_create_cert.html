OpenSSL - Create Certificates

openssl req -new -x509 -out cert.pem -keyout key.pem

or

openssl genrsa -des3 -out key.pem 2048 // create private key with pwd
openssl genrsa -out key.pem 2048 // create private key without pwd
openssl req -new -key key.pem -out cert.csr // create a cert request
openssl req -new -x509 -key key.pem -out cert.pem -days 3650 // create self-signed cert(signed by key.pem)
