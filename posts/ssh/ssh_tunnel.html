SSH Tunnel

  • Local Port Forwarding: listening on ssh client, forwarding from ssh server 
        # -f is running in background, optinal
    ssh -NfL [bind_address:]port:host:hostport] johnson@jserver
  • Remote Port Forwarding: listening on ssh server, forwarding from ssh client 
        # -f is running in background, optinal
    ssh -NfR [bind_address:]port:host:hostport] johnson@jserver
  • Dynamic Port Forwarding: listening on ssh client, forwarding from ssh server, target host:port is determined by SOCKS4/SOCKS5 (useful with Google 'Proxy SwitchyOmega' extension or 'tsocks' command line tool) 
        # -f is running in background, optinal
    ssh -NfD [bind_address:]port johnson@jserver

privoxy: turn SOCKS4/SOCKS5 into HTTP proxy

yum install epel-release
yum install privoxy

# vim /etc/privoxy/conf
forward-socks5 / 127.0.0.1:1080 .

systemctl start privoxy.service
